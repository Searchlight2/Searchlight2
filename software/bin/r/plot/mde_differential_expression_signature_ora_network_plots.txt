####---- Differential Expression Signatures ORA Enriched Gene Sets (Network Plots) ----####

node_colours = default_two_tone_network_node_colours
plot_height = 2000
plot_width = 2000
node_outer_max_size = 10
node_inner_max_size = 8
node_label_size = 3

<*per_ora*>
for (signature in 1:nrow(signature_summary))
{
  edges = read.table(file=paste0("data/statistical_analysis/differential_expression_signature/over_representation_analysis/<*type*>/signature_",signature,"/network_data/enriched_gene_sets_edges.csv"), header=TRUE, sep='\t', quote='',check.names = TRUE)
  nodes = read.table(file=paste0("data/statistical_analysis/differential_expression_signature/over_representation_analysis/<*type*>/signature_",signature,"/network_data/enriched_gene_sets_nodes.csv"), header=TRUE, sep='\t', quote='',check.names = TRUE)

  ggp = make_ora_network_plot(edges,nodes,node_colours,node_outer_max_size,node_inner_max_size,node_label_size, TRUE)
  save_plot(ggp,plot_height,plot_width,paste0("<*path*>plots/differential_expression_signature/signature_ora_networks/<*type*>/<*/path*>signature_",signature,"_enriched_gene_sets_network.png"))

  ggp = make_ora_network_plot(edges,nodes,node_colours,node_outer_max_size,node_inner_max_size,node_label_size, FALSE)
  save_plot(ggp,plot_height,plot_width,paste0("<*path*>plots/differential_expression_signature/signature_ora_networks/<*type*>/<*/path*>signature_",signature,"_enriched_gene_sets_network_unlabelled.png"))
}
